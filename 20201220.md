# Leetcode Everyday 2020-12-20

## Leetcode 200 Number of Islands
#DFS #BFS #Union Find

Given an m x n 2d grid map of '1's (land) and '0's (water), return the number of islands.

An island is surrounded by water and is formed by connecting adjacent lands horizontally or vertically. You may assume all four edges of the grid are all surrounded by water.

### Method 1: DFS
```java
class Solution {
    public int numIslands(char[][] grid) {
        int num = 0;
        // Special case
        if (grid == null || grid.length == 0) {
            return num;
        }

        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == '1') {
                    dfs(grid, i, j);
                    num++;
                }
            }
        }
        return num;
    }

    private void dfs(char[][] grid, int row, int col) {
        if (row < 0 || row >= grid.length || col < 0 || col >= grid[0].length || grid[row][col] == '0') {
            return;
        }
        grid[row][col] = '0';
        dfs(grid, row, col + 1);
        dfs(grid, row, col - 1);
        dfs(grid, row + 1, col);
        dfs(grid, row - 1, col);
    }
}
```

### Method 2: BFS
```java
class Solution {
    public int numIslands(char[][] grid) {
        int num = 0;
        // Special case
        if (grid == null || grid.length == 0) {
            return num;
        }

        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == '1') {
                    bfs(grid, i, j);
                    num++;
                }
            }
        }
        return num;
    }

    private void bfs(char[][] grid, int row, int col) {
        if (row < 0 || row >= grid.length || col < 0 || col >= grid[0].length || grid[row][col] == '0') {
            return;
        }
        grid[row][col] = '0';
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[] {row, col});
        while (!queue.isEmpty()) {
            int[] firstPair = queue.remove();
            int r = firstPair[0];
            int c = firstPair[1];
            if (r >= 0 && r < grid.length && c + 1 >= 0 && c + 1 < grid[0].length && grid[r][c + 1] == '1') {
                queue.add(new int[] {r, c + 1});
                grid[r][c + 1] = '0';
            }
            if (r >= 0 && r < grid.length && c - 1 >= 0 && c - 1 < grid[0].length && grid[r][c - 1] == '1') {
                queue.add(new int[] {r, c - 1});
                grid[r][c - 1] = '0';
            }
            if (r + 1 >= 0 && r + 1 < grid.length && c >= 0 && c < grid[0].length && grid[r + 1][c] == '1') {
                queue.add(new int[] {r + 1, c});
                grid[r + 1][c] = '0';
            }
            if (r - 1 >= 0 && r - 1< grid.length && c >= 0 && c < grid[0].length && grid[r - 1][c] == '1') {
                queue.add(new int[] {r - 1, c});
                grid[r - 1][c] = '0';
            }
        }
    }
}
```

### Method 3: Union Find (AKA, disjoint set)
```java
class Solution {
    public int numIslands(char[][] grid) {
        int num = 0;
        // Special case
        if (grid == null || grid.length == 0) {
            return num;
        }

        UnionFind uf = new UnionFind(grid);
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[0].length; j++) {
                if (grid[i][j] == '1') {
                    unionFindHelper(uf, grid, i, j);
                }
            }
        }

        return uf.getCount();
    }

    private void unionFindHelper(UnionFind uf, char[][] grid, int r, int c) {
        if (r < 0 || r >= grid.length || c < 0 || c >= grid[0].length || grid[r][c] == '0') {
            return;
        }
        grid[r][c] = '0';
        int numCol = grid[0].length;
        int currIndex = r * numCol + c;
        if (r >= 0 && r < grid.length && c + 1 >= 0 && c + 1 < grid[0].length && grid[r][c + 1] == '1') {
                uf.union(currIndex, r * numCol + (c + 1));
            }
            if (r >= 0 && r < grid.length && c - 1 >= 0 && c - 1 < grid[0].length && grid[r][c - 1] == '1') {
                uf.union(currIndex, r * numCol + (c - 1));
            }
            if (r + 1 >= 0 && r + 1 < grid.length && c >= 0 && c < grid[0].length && grid[r + 1][c] == '1') {
                uf.union(currIndex, (r + 1) * numCol + c);
            }
            if (r - 1 >= 0 && r - 1< grid.length && c >= 0 && c < grid[0].length && grid[r - 1][c] == '1') {
                uf.union(currIndex, (r - 1) * numCol + c);
            }
    }

    // Helper class
    class UnionFind {
        int count; // # of connected components;
        int[] parent;
        int[] rank;

        public UnionFind(char[][] grid) {
            count = 0;
            int numRow = grid.length;
            int numCol = grid[0].length;
            parent = new int[numRow * numCol];
            rank = new int[numRow * numCol];
            for (int i = 0; i < numRow; i++) {
                for (int j = 0; j < numCol; j++) {
                    if (grid[i][j] == '1') {
                        parent[i * numCol + j] = i * numCol + j;
                        count++;
                    }
                    rank[i * numCol + j] = 0;
                }
            }
        }

        // find root
        private int find(int i) {
            if (parent[i] != i) {
                parent[i] = find(parent[i]);
            }
            return parent[i];
        }

        // union two connected components
        private void union(int x, int y) {
            int rootX = find(x);
            int rootY = find(y);
            if (rootX != rootY) {
                if (rank[rootX] > rank[rootY]) {
                    parent[rootY] = rootX;
                } else if (rank[rootX] < rank[rootY]) {
                    parent[rootX] = rootY;
                } else {
                    parent[rootY] = rootX;
                    rank[rootX]++;
                }
                count--;
            }
        }

        public int getCount() {
            return count;
        }
    }
}
```
